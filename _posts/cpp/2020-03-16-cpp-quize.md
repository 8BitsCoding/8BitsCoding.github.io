// string 들어오면 ,기준으로 파싱
vector<int> convertToVector(const string &str) {
    vector<int> result;
    istringstream iss(str);
    string line;
    while (getline(iss, line, ',')) {
      int value;
      istringstream issl(line);
      issl >> value;
      result.push_back(value);
    }
    return result;
}
