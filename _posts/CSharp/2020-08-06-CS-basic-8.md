```csharp
static void Main()
{
    // string은 참조형이기에 null이 가능
    string s1 = "Hello";
    string s2 = null;

    // int는 값이기에 null을 넣을 순 없다.
    int n1 = 10;
    // int n2 = null;

    Nullable<int> n3 = 10;
    Nullable<int> n4 = null;    // nullable과 함께라면 okay

    int? n5;    // int? = Nullable<int>와 동일한 표현이다
}
```

언제쓰나

```csharp
public static Nullable<int> foo()
{
    if(실패)
        return null;        // okay
}
```

```csharp
static void Main()
{
    int? n1 = 10;
    int a1 = 20;

    n1 = a1;    // int? to int -> okay
    a1 = n1;    // error
    a1 = (int)n1;    // okay
}
```
